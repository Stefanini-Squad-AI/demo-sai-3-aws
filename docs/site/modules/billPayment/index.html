Validation Rules:
- Account must be in ACTIVE status
- Payment amount must be > 0 and <= current balance
- Payment method must be valid (BANK_ACCOUNT, DEBIT_CARD, CHECK)
- Payment date cannot be in the past

üí° Implementation Examples

Using Bill Payment Hook

import { useBillPayment } from '~/features/billPayment/hooks/useBillPayment';

function PaymentForm({ accountId }) {
  const { submitPayment, loading, error } = useBillPayment();
  
  const handleSubmit = async (data) => {
    const result = await submitPayment({
      accountId,
      paymentAmount: data.amount,
      paymentMethod: 'BANK_ACCOUNT'
    });
    
    if (result.success) {
      alert(`Payment confirmed: ${result.confirmationNumber}`);
    }
  };
  
  return <PaymentForm onSubmit={handleSubmit} loading={loading} />;
}

Payment Confirmation Display

import { PaymentConfirmation } from '~/features/billPayment/PaymentConfirmation';

function ConfirmationPage({ paymentResult }) {
  return (
    <PaymentConfirmation
      confirmationNumber={paymentResult.confirmationNumber}
      amount={paymentResult.processedAmount}
      newBalance={paymentResult.newBalance}
      paymentDate={paymentResult.paymentDate}
      onPrint={() => window.print()}
    />
  );
}
