ğŸ’¡ Implementation Examples

Card List with Pagination

import { useCreditCardList } from '~/features/creditCard/hooks/useCreditCardList';

function CardListPage() {
  const { cards, loading, page, totalPages, setPage } = useCreditCardList({
    pageSize: 20,
    status: 'ACTIVE'
  });
  
  if (loading) return <LoadingSpinner />;
  
  return (
    <div>
      <CardTable cards={cards} />
      <Pagination 
        current={page} 
        total={totalPages} 
        onChange={setPage} 
      />
    </div>
  );
}

Update Card Status

import { useCreditCardUpdate } from '~/features/creditCard/hooks/useCreditCardUpdate';

function SuspendCardButton({ cardNumber }) {
  const { updateCard, loading } = useCreditCardUpdate();
  
  const handleSuspend = async () => {
    await updateCard({
      cardNumber,
      newStatus: 'SUSPENDED',
      reasonCode: 'FRAUD_SUSPECTED',
      notes: 'Customer reported suspicious activity'
    });
  };
  
  return (
    <button onClick={handleSuspend} disabled={loading}>
      Suspend Card
    </button>
  );
}
