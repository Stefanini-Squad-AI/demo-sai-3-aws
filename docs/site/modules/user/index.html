Create New User

import { UserAddScreen } from '~/features/user/UserAddScreen';

function CreateUserPage() {
  const handleSuccess = (userId) => {
    console.log('User created:', userId);
    navigate('/admin/users');
  };
  
  return (
    <UserAddScreen
      onSuccess={handleSuccess}
      availableRoles={['ADMIN', 'MANAGER', 'CSR', 'VIEWER']}
    />
  );
}

Update User Role

import { UserUpdateScreen } from '~/features/user/UserUpdateScreen';

function EditUserPage({ userId }) {
  const handleUpdate = async (data) => {
    // Update user details and role
    await updateUser({
      userId,
      fullName: data.name,
      userType: data.role
    });
  };
  
  return <UserUpdateScreen userId={userId} onSubmit={handleUpdate} />;
}
