üí° Implementation Examples

Using Account View Hook

import { useAccountView } from '~/features/account/hooks/useAccountView';

function AccountDetails() {
  const { account, loading, error } = useAccountView('ACC123456');
  
  if (loading) return <LoadingSpinner />;
  if (error) return <ErrorMessage error={error} />;
  
  return (
    <div>
      <h2>{account.accountHolder}</h2>
      <p>Balance: ${account.currentBalance}</p>
      <p>Available: ${account.availableCredit}</p>
    </div>
  );
}

Account Update Form

import { useAccountUpdate } from '~/features/account/hooks/useAccountUpdate';

function UpdateAccount({ accountId }) {
  const { updateAccount, loading } = useAccountUpdate();
  
  const handleSubmit = async (data) => {
    await updateAccount({
      accountId,
      accountHolder: data.name,
      contactEmail: data.email,
      contactPhone: data.phone
    });
  };
  
  return <AccountForm onSubmit={handleSubmit} loading={loading} />;
}
